1. Initialize variables:
   - Initialize `min_diff` to a large value (e.g., 65535, the maximum value for a 16-bit integer).
   - Initialize `max_diff` to 0.

2. Loop through each element `i` in the array (from 0 to n-1):
   - Loop through each element `j` in the array where j > i (to avoid repeated comparisons and self-comparison):
     - Calculate the difference: `diff = abs(array[i] - array[j])`
     - Update `min_diff` if `diff < min_diff`
     - Update `max_diff` if `diff > max_diff`

3. Store the results:
   - Calculate `min_diff_MSB` = `min_diff` >> 8 (Right shift `min_diff` by 8 bits to extract the most significant byte)
   - Calculate `min_diff_LSB` = `min_diff` & 255 (Bitwise AND with 255 to extract the least significant byte)
   - Calculate `max_diff_MSB` = `max_diff` >> 8
   - Calculate `max_diff_LSB` = `max_diff` & 255

   - Store these values in memory:
     - mem[66] = `min_diff_MSB`
     - mem[67] = `min_diff_LSB`
     - mem[68] = `max_diff_MSB`
     - mem[69] = `max_diff_LSB`

    MOV RC2 0 // inner loop counter

    MOV RDS 0

    _inLoop:
        LEA RAS 
        LEA RBS
        LEA RCS
        LEA RDS
        TOG // mode 1
        ADD RAS 64
        ADD RAS 64
        ADD RAS 64
        DEC RAS
        MOV RBS 1
        ADD RCS 64
        ADD RCS 64
        ADD RCS 64
        DEC RCS
        DEC RCS
        MOV RDS 1
        TOG // mode 0
        MOV RAS [RAS]  // load progress1 MSB1
        ADD RBS RAS     // LSB1
        MOV RCS [RCS]  // load progress 2
        ADD RCS RAS     // MSB2
        ADD RDS RCS     // LSB2    
        
        MOV RAS [RAS]
        MOV RBS [RBS]
        MOV RCS [RCS]
        MOV RDS [RDS] 

        CMP RAS RCS     // check who is larger
        JLT _lessThan1  // RAS < RCS
        SUB RAS RCS     // MSB1 dist

        CMP RBS RDS
        JLT _lessThan2 // RBS < RDS
        SUB RBS RDS  // RAS = msb dist, RBS = lsb dist
        JUMP _endcomp1

        _lessThan1:
            SUB RCS RAS   // RCS = MSB
            CMP RDS RBS     
            JLT _lessThan3 // RDS < RBS
            SUB RDS RBS     // RCS = msb dist, RDS = lsb dist
            _endcomp2

        _lessThan2:
            SUB RDS RBS // RAS = msb dist, RDS = lsb dist
            _endcomp3

        _lessThan3:
            SUB RBS RDS
            TOG // mode 1
            DEC RBS // RCS = msb dist, RBS = lsb dist
            _endcomp4

        _endComp1:
            LEA RCS
            LEA RDS
            TOG // mode 1
            ADD RCS 64
            ADD RCS 64
            ADD RCS 64 // mem[192] = LSB dist

            ADD RDS 64
            ADD RDS 64
            ADD RDS 64 
            ADD RDS 1 // mem[193] = MSB dist
            TOG // mode 0
            MOV [RCS] RBS

        _endComp2:

            LEA RBS
            TOG // mode 1
            ADD RAS 64
            ADD RAS 64
            ADD RAS 64 // mem[192] = LSB dist

            ADD RBS 64
            ADD RBS 64
            ADD RBS 64 
            ADD RBS 1 // mem[193] = MSB dist
            TOG // mode 0
            MOV [RAS] RDS
            MOV [RBS] RCS

        _endComp3:
            LEA RCS
            LEA RBS
            TOG // mode 1
            ADD RCS 64
            ADD RCS 64
            ADD RCS 64 // mem[192] = LSB dist

            ADD RBS 64
            ADD RBS 64
            ADD RBS 64 
            ADD RBS 1 // mem[193] = MSB dist
            TOG // mode 0
            MOV [RCS] RDS
            MOV [RBS] RAS

        _endComp4:

            TOG // mode 1
            ADD RAS 64
            ADD RAS 64
            ADD RAS 64 // mem[192] = LSB dist

            ADD RDS 64
            ADD RDS 64

            MOV [RDS] RCS

        _compend:
            LEA RAS
            LEA RBS
            LEA RCS
            LEA RDS
            TOG // mode 1
            ADD RAS 64
            ADD RAS 1
            ADD RAS 1
            ADD RBS 1

            ADD RCS 64
            ADD RCS 64
            ADD RDS 64
            ADD RDS 64

            MOV RBS [RBS]  // min_diff_LSB
            MOV RCS [RCS]  // LSB dist
            MOV RDS [RDS]  // MSB dist

            CMP RDS RAS  
            JLT _minLess // MSB dist < min_diff_MSB
            SUB RAS RDS
            JZ _minequal
            JUMP _checkMax

        _minequal:
            CMP RCS RBS
            JLT _minLess
            JUMP _checkMax

        _minLess:
            LEA RAS
            LEA RBS
            TOG // mode 1
            ADD RAS 64
            ADD RAS 1
            ADD RAS 1
            ADD RBS 1
            TOG // mode 0

            MOV [RBS] RCS

            ADD RAS 4
            ADD RBS 1

            MOV RBS [RBS]  // min_diff_LSB
            CMP RAS RDS
            JLT _maxLess
            SUB RAS RDS
            JZ _maxequal
            JUMP _continue

        _maxequal:
            CMP RBS RCS

            MOV [RBS] RCS
